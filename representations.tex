\section{Representations}

\begin{definition}[Representation]
\label{rep.def}
Given a functor $H : \cat{C} \to \catSet$, a \emph{representation} of $H$ is a pair of 
\In{enumerate}{
    \item an object $R$ in \cat{C}
    \item a natural bijection $(\tau_X : HX \cong \cat{C}(R,X)) _ X$
}
\end{definition}
The bijectivity can be expressed using the cute boxes
\In{align*}{
    \strdgathered{
        \strdfork{
            \strdmid
        }
        ,\strdfork{
            \strdtubestartline
            ,\strdtube\strdtubemid
            ,\strdwidecell{1}{1}{a}
        }
    }
    =
    \strdgathered{
        \strdfork{
            \strdmid
            ,\strdcell{f}
            ,\strdmid,\strdrlabel{R}
        }
        ,\strdfork{
            \strdtubestartline
            ,\strdtube\strdtubemid
            ,\strdtwincell{\tau}
            ,\strdtubeof{strd frame}\strdtubeclose
        }
    } \iff
    \strdgathered{
        \strdmid
        ,\strdfork{
            \strdmid
        }
        ,\strdfork{
            \strdtubestart
            ,\strdtwincell{\tau}
            ,\strdtube\strdtubemid
            ,\strdwidecell{1}{1}{a}
        }
        ,\strdfork{
            \strdtubestart
            ,\strdtubeof{strd frame}{
                \strdtubesqcap{1}{1}
                ,\strdyscale{0.7}\strdtubeclose
            }
            ,\strdmid,\strdrlabel{R}
        }
    }
    =\,
    \strdgathered{
        \strdmid
        ,\strdcell{f}
        ,\strdmid,\strdrlabel{R}
    }
}
thanks to the naturality.

The following allows you to denote the represention of $H$ as %
$\text{rep}_X HX$ or briefly $\text{rep}H$.

\begin{definition}
Given a functor $H : \cat{B}\catop \times \cat{A} \to \catSet$, %
a natural bijection of the form
\In{align*}{
    (\tau_X : H(B,X) \cong \cat{A}(A,X)) _{X}
}
can be expressed by
\In{align*}{
    &
    \strdgathered{
        \strdmid
        ,\strdcell{f}
        ,\strdmid,\strdrlabel{A}
    } = 
    \strdgathered{
        \strdfork{
            \strdtubestart
            ,\strdtubemid
            ,\strdat{Left}{\strdrlabel{B}}
            ,\strdat{Right Start}{ \strdinv\strdmid }
            ,\strdat{Left Start}{ \strdhleft }

            \strdtubestart

\end{definition}

\begin{proposition}
\label{p.rep.i.f}
Given a functor $H : \cat{B}\catop \times \cat{A} \to \catSet$ and a family of representations
\In{align*}{
    ( ( \tau_X^B : H(B,X) \cong \cat{A}(S_B,X) )_X )_ B
}
, there exists a unique infrafunctor $S : \cat{B} \to \cat{A}$ such that
\In{itemize}{
    \item $SB = S_B$
    \item $\tau$ is natural in $B$
}
Furthermore, $S$ is functorial.
\end{proposition}

\begin{strdproof}
Define $S$ as
\In{align*}{
    \strdgathered{
        \strdfork{
            \strdtubestart
            ,\strdtubemid
            ,\strdat{Left}{\strdrlabel{B}}
            ,\strdat{Right Start}{ \strdinv\strdmid,\strdrlabel{S_{B\myprime} }}
            ,\strdat{Left Start}{ \strdhleft }
            ,\strdat{Left End}{\strdcell{\strdupsidedown{\catph{?}}}}
            ,\strdtubemid
            ,\strdat{Left}{\strdrlabel{B\myprime}}
            ,\strdat{Left End}{\strdhleft}
            ,\strdat{Right End}{\strdmid}
        }
        ,\strdfork{
            \strdtubestart
            ,\strdtubeextend{1}{1}
            ,\strdfork{
                \strdtubeof{strd frame}{
                    \strdtubesqcap{1}{1}
                    ,\strdtubemid
                    ,\strdtubemid
                    ,\strdyscale{0.7}\strdtubeclose
                }
                ,\strdmid,\strdrlabel{S_B}
            }
            ,\strdtwincell{\tau}
            ,\strdtube\strdtubemid
            ,\strdtube\strdtubemid
            ,\strdtwincell{\tau\strdrlap\myprime}
            ,\strdtubeof{strd frame}\strdtubeclose
        }
    }
}
\end{strdproof}
