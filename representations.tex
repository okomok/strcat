\section{Representations}

\definitionof{Representation}{
    Given a functor $H : \cat{C} \to \catSet$, a \emph{representation} of $H$ is a pair of 
    \In{enumerate}{
        \item an object $R$ in \cat{C}
        \item a natural bijection $(\tau_X : HX \cong \cat{C}(R,X)) _ X$
    }
}

The bijectivity can be expressed using weird boxes
\In{align*}{
    \strdgathered{
        \strdfork{
            \strdmid
        }
        ,\strdfork{
            \strdtubestartline
            ,\strdtube\strdtubemid
            ,\strdwidecell{1}{1}{a}
        }
    }
    =
    \strdgathered{
        \strdfork{
            \strdmid
            ,\strdcell{f}
            ,\strdmid,\strdrlabel{R}
        }
        ,\strdfork{
            \strdtubestartline
            ,\strdtube\strdtubemid
            ,\strdtwincell{\tau}
            ,\strdtubeof{strd frame}\strdtubeclose
        }
    } \iff
    \strdgathered{
        \strdmid
        ,\strdfork{
            \strdmid
        }
        ,\strdfork{
            \strdtubestart
            ,\strdtwincell{\tau}
            ,\strdtube\strdtubemid
            ,\strdwidecell{1}{1}{a}
        }
        ,\strdfork{
            \strdtubestart
            ,\strdtubeof{strd frame}{
                \strdtubesqcap{1}{1}
                ,\strdyscale{0.7}\strdtubeclose
            }
            ,\strdmid,\strdrlabel{R}
        }
    }
    =\,
    \strdgathered{
        \strdmid
        ,\strdcell{f}
        ,\strdmid,\strdrlabel{R}
    }
}
thanks to the naturality. The following proposition allows you to denote the represention of $H$ as 
\In{math}{
    \text{rep}H
}.

\propositionof{Uniqueness of Representations}{
    Representations are unique up to unique isomorphism.
}

\In{strdproof}{
    Let $(R\myprime,\tau\myprime)$ be another representation. %
    By the variant of \cref{Yoneda Principle}, 
    \In{align*}{
        &
        \strdgathered{
            \strdmid
            ,\strdcell{\catph{?}}
            ,\strdmid,\strdrlabel{R\myprime}
            ,\strdcell{\varphi}
            ,\strdmid,\strdrlabel{R}
        } \,=\,
        \strdgathered{
            \strdmid
            ,\strdfork{
                \strdmid
                ,\strdcell{\catph{?}}
                ,\strdmid,\strdrlabel{R\myprime}
            }
            ,\strdtubestart
            ,\strdfork{
                \strdtwincell{\tau}
                ,\strdtube\strdtubemid
                ,\strdtwincell{\tau\strdrlap\myprime}
                ,\strdtubeof{strd frame}\strdtubeclose
            }
            ,\strdfork{
                \strdtubeof{strd frame}{
                    \strdtubesqcap{1}{1}
                    ,\strdtubemid
                    ,\strdyscale{0.7}\strdtubeclose
                }
                ,\strdmid,\strdrlabel{R}
            }
        } : \text{invertible} \\
        \iff\, &
        \strdgathered{
            \strdmid,\strdrlabel{R\myprime}
            ,\strdcell{\varphi}
            ,\strdmid,\strdrlabel{R}
        } \,=\,
        \strdgathered{
            \strdmid,\strdrlabel{R\myprime}
            ,\strdfork{
                \strdmid
                ,\strdmid
            }
            ,\strdtubestart
            ,\strdfork{
                \strdtwincell{\tau}
                ,\strdtube\strdtubemid
                ,\strdtwincell{\tau\strdrlap\myprime}
                ,\strdtubeof{strd frame}\strdtubeclose
            }
            ,\strdfork{
                \strdtubeof{strd frame}{
                    \strdtubesqcap{1}{1}
                    ,\strdtubemid
                    ,\strdyscale{0.7}\strdtubeclose
                }
                ,\strdmid,\strdrlabel{R}
            }
        } : \text{invertible} \\
    }
}

\In{definition}{
    Given a functor $H : \cat{B}\catop \times \cat{A} \to \catSet$, %
    a natural bijection of the form
    \In{align*}{
        (\tau_X : H(B,X) \cong \cat{A}(A,X)) _{X}
    }
    can be expressed by
    \In{align*}{
        &
        \strdgathered{
            \strdmid
            ,\strdcell{f}
            ,\strdmid,\strdrlabel{A}
        } = 
        \strdgathered{
            \strdfork{
                \strdtubestart
                ,\strdtubemid
                ,\strdat{Left}{\strdrlabel{B}}
                ,\strdat{Right Start}{ \strdinv\strdmid }
                ,\strdat{Left Start}{ \strdhleft }
            }
            ,\strdfork{
                \strdtubestart
                ,\strdtubeextend{1}{1}
                ,\strdfork{
                    \strdtubeof{strd frame}{
                        \strdtubesqcap{1}{1}
                        ,\strdyscale{0.7}\strdtubeclose
                    }
                    ,\strdmid,\strdrlabel{A}
                }
                ,\strdtwincell{\tau}
                ,\strdtube\strdtubemid
                ,\strdwidecell{2}{2}{a}
            }
        }\\
        \iff &
        \strdgathered{
            \strdtubestart
            ,\strdfork{
                \strdtubemid
                ,\strdat{Left}{\strdrlabel{B}}
                ,\strdat{Right End}{\strdcell{f},\strdmid,\strdrlabel{A}}
                ,\strdat{Left End}{\strdhleft}
            }
            ,\strdfork{
                \strdtubeextend{1}{1},\strdcutoff\strdtubebound
                ,\strdtube\strdtubemid
                ,\strdtwincell{\tau}
                ,\strdtubeof{strd frame}\strdtubeclose
            }
        } =
        \strdgathered{
            \strdtubestart
            ,\strdfork{
                \strdtubemid
                ,\strdat{Left}{\strdrlabel{B}}
            }
            ,\strdfork{
                \strdtubeextend{1}{1},\strdcutoff\strdtubebound
                ,\strdtubemid
                ,\strdwidecell{2}{2}{a}
            }
        }
    }
}

\propositionof{Parameterized Representations}{
    Let $H : \cat{B}\catop \times \cat{A} \to \catSet$ be a functor. Given a family of objects
    \In{math}{
        (SB)_{B}
    }
    and a family of representations
    \In{align*}{
        (( \tau_X^B : H(B,X) \cong \cat{A}(SB,X) )_X )_ B
    }
    % \In{enumerate}{
    %     \item a family of objects $(SB \in  \catOb A)_{B \in \catOb B}$
    %     \item a family of representations $ ( ( \tau_X^B : H(B,X) \cong \cat{A}(SB,X) )_X )_ B$
    % }
    there exists a unique family
    \In{align*}{
        (S(f) \in \cat A(SB, SB\myprime))_{f \in \cat B(B, B\myprime)}  
    }
    such that $\tau$ is natural in $B$. Furthermore, $S$ is functorial.
}

\In{strdproof}{
    Define $S$ as
    \In{align*}{
        \strdgathered{
            \strdfork{
                \strdtubestart
                ,\strdtubemid
                ,\strdat{Left}{\strdrlabel{B}}
                ,\strdat{Right Start}{ \strdinv\strdmid,\strdrlabel{S_{B\myprime} }}
                ,\strdat{Left Start}{ \strdhleft }
                ,\strdat{Left End}{\strdcell{\strdupsidedown{\catph{?}}}}
                ,\strdtubemid
                ,\strdat{Left}{\strdrlabel{B\myprime}}
                ,\strdat{Left End}{\strdhleft}
                ,\strdat{Right End}{\strdmid}
            }
            ,\strdfork{
                \strdtubestart
                ,\strdtubeextend{1}{1}
                ,\strdfork{
                    \strdtubeof{strd frame}{
                        \strdtubesqcap{1}{1}
                        ,\strdtubemid
                        ,\strdtubemid
                        ,\strdyscale{0.7}\strdtubeclose
                    }
                    ,\strdmid,\strdrlabel{S_B}
                }
                ,\strdtwincell{\tau}
                ,\strdtube\strdtubemid
                ,\strdtube\strdtubemid
                ,\strdtwincell{\tau\strdrlap\myprime}
                ,\strdtubeof{strd frame}\strdtubeclose
            }
        }
    }
}
